import java.util.HashSet;
import java.util.Map;
import java.util.Scanner;
import java.util.Set;
import java.util.TreeMap;

public class program {
	
	public static void printMap(Map<String, Set<String>> map) {
		map.forEach((k,v) -> System.out.println(k + " -> " + v ));
	}

	public static void main(String[] args) {
		Scanner reader = new Scanner(System.in);
		String str = reader.nextLine();
		Set<String> syn = new HashSet<>();
		

		Map<String, Set<String>> map = new TreeMap<>();
		syn.add("Good Morning");
		syn.add("Shubhodaya");
		syn.add("Shubhoday");
		syn.add("Bonjour");
		map.put("Good Morning", syn);
		
		if(str.isEmpty()) {
			printMap(map);
			return;
		}
		if(!map.containsKey(str)) {
			System.out.println("No such word found. Would you like to add it to the dictionary ? (Y/N)");
			char resp = reader.next().charAt(0);
			if(resp == 'Y') {
				Set<String> userSyn = new HashSet<>();
				System.out.println("Please enter the synonym ");
				String syn1 = null;
				reader.nextLine();
				while(true) {
					syn1 = reader.nextLine();
					if(syn1.equals("")) {
						break;
					}else {
						userSyn.add(syn);
					}
				}
				
				map.put(str,userSyn);
			}
			
		}
		
		printMap(map);
		
				
	}

}
